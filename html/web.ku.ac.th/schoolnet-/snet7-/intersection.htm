+อินเตอร์เซคชั่น (intersection)
   อินเตอร์เซคชั่นของเซต a และ b คือเซตที่ประกอบด้วยสมาชิกซึ่งเป็นสมาชิกของทั้งเซต a และเซต bเขียนแทนด้วย a ว b
        นั่นคือ   aวb = {x/xฮa และ x ฮb }

 ตัวอย่างที่ 1 กำหนด u = {1, 2, 3, ... ,10} , a = {1, 3, 5, 7} และ b = {2, 4, 6, 8}
 วิธีทำ\a ว b = { } = ฦ
            b ว a = { } = ฦ

 ตัวอย่างที่ 2 กำหนด u = {1, 2, 3, ... ,10} , a = {2, 4, 6, 8} และ b = {4, 5, 7}
 วิธีทำ\a ว b = { 4}
           b ว a = { 4}

ตัวอย่างที่ 3 กำหนด u = {1, 2, 3, ... ,10} , a = {1, 2,3} และ b = {1, 2, 3, 4}
 วิธีทำ\a ว b = {1, 2, 3 } = a
           b ว a = {1, 2, 3 } = a

aฬ b : a ว b = a = b ว a

ตัวอย่างที่ 4   กำหนด u = {a, b, c, d, e, f} , a = {a, b, c, d, e } และ b = {a, d, e}
วิธีทำ\a ว b = {a, d, e}
           b ว a = {a, d, e}
           bฬ a : a ว b = b = b ว a

ตัวอย่างที่ 5 กำหนด a ={a, b, c}
            \ a ว a = {a, b, c} = a

ตัวอย่างที่ 6 กำหนด a ={a, b, c}
            \ a วฦ = ฦว a = { } = ฦ

ตัวอย่างที่ 7 กำหนด u = {1, 2, 3, ... ,10} และ a = {1, 2,3}
            \ a ว u = a ว u = {1, 2, 3} = a

สรุปจากตัวอย่าง
a ว b = b ว a
ถ้า aฬ b แล้ว a ว b = b ว a = a
ถ้า bฬ a แล้ว a ว b = b ว a = b
a ว a = a
a วฦ =ฦ ว a = a
a ว u = a ว u = u
