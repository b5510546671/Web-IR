program primelist1;
var
n,q,max:extended;
begin
write(' max = ? ');
(*  prompts you for the maximum:  *)
(*  all primes  < max   are printed out  *)
readln(max);
write('2');
n:=3;
while n<=max do
begin
q:=3;
qdividesn:=false;
(* initialization for the while loop *)
while (q*q<=n) and not(qdividesn)  do
 begin
 qdividesn:=(n=q*int(n/q));
 q:=q+2;
 end;
if  not(qdividesn)  then  write(',',n:0:0);
n:=n+2;
end;
end.
